def start_session():

[docs]    """
    Start GTK session
    """

    from gi.repository import Gtk

    global _gtk_session

    import signal

    if not _gtk_session:
        logging.debug("start_session")

        signal.signal(signal.SIGINT, signal.SIG_DFL)

        _gtk_session = True

        MainWindow()

        try:
            Gtk.main()
        except:
            logging.exception("Exception while running Gtk.main()")
            stop_session()

        _gtk_session = None
