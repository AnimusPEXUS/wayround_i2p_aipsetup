def load_tags_from_fs():


[docs]
    tag_db = org.wayround.aipsetup.dbconnections.tag_db()

    file_name = org.wayround.aipsetup.config.config['tags']

    try:
        f = open(file_name, 'r')
    except:
        logging.exception("Couldn't open file `{}'".format(file_name))
    else:
        try:
            txt = f.read()

            d = json.loads(txt)

            tag_db.clear()

            keys = list(d.keys())
            keys.sort()

            count = len(keys)
            num = 0
            for i in keys:
                num += 1

                if num == 0:
                    perc = 0
                else:
                    perc = float(100) / (float(count) / float(num))
                org.wayround.utils.file.progress_write('    {:6.2f}%'.format(perc))
                tag_db.set_tags(i, d[i])

            org.wayround.utils.file.progress_write_finish()
        finally:

            f.close()

    return
